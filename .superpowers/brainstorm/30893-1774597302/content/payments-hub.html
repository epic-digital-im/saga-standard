X402Client Service

Payment client that wraps all hub API calls
┌─────────────────────────────────────────────────────────────┐
│  class X402Client                                            │
│                                                              │
│  constructor(wallet: AbstractWallet, facilitatorUrl: string) │
│                                                              │
│  // Core method — wraps any fetch call with x402 handling    │
│  async fetchWithPayment(url, options): Response              │
│    1. Make initial request                                   │
│    2. If 402 → parse payment options                         │
│    3. Emit 'payment-required' event → triggers PaymentSheet  │
│    4. Wait for user confirmation                             │
│    5. Sign USDC transfer with wallet                         │
│    6. Submit to facilitator for settlement proof             │
│    7. Retry request with PAYMENT-RESPONSE header             │
│    8. Return final response                                  │
│                                                              │
│  // Convenience methods                                      │
│  async provision(hubUrl, plan, name): ProvisionResult        │
│  async renew(hubUrl, tenantId, plan): RenewResult            │
│  async checkStatus(hubUrl, tenantId): StatusResult           │
│  async listPlans(hubUrl): Plan[]                             │
│                                                              │
│  // Payment history (local)                                  │
│  getPaymentHistory(): PaymentEvent[]                         │
│  getActiveSubscriptions(): Subscription[]                    │
│                                                              │
│  // Events                                                   │
│  on('payment-required', handler)  → show PaymentSheet        │
│  on('payment-settled', handler)   → update UI                │
│  on('payment-failed', handler)    → show error               │
│                                                              │
└─────────────────────────────────────────────────────────────┘

  Supported networks: Base (EVM) + Solana
  Settlement asset: USDC
  Facilitator: https://x402.org/facilitator
